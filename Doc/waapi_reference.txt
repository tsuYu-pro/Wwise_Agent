## WAAPI Connection Setup

WAAPI uses WAMP (Web Application Messaging Protocol) over WebSocket. Default endpoint: ws://127.0.0.1:8080/waapi

Prerequisites:
- Wwise Authoring Application must be running
- WAAPI must be enabled: Project > Project Settings > WAMP > Enable WAAPI
- Default port: 8080 (configurable in Project Settings)

Python connection using waapi-client:
```python
from waapi import WaapiClient

# Connect to Wwise
client = WaapiClient()

# Or specify custom URL
client = WaapiClient(url="ws://127.0.0.1:8080/waapi")

# Always disconnect when done
client.disconnect()
```

## Creating Objects with ak.wwise.core.object.create

Create Wwise objects in the project hierarchy.

Basic sound creation:
```python
result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
        "type": "Sound",
        "name": "MyNewSound",
        "onNameConflict": "rename"  # or "replace", "fail"
    }
)
# Returns: {"id": "{GUID}", "name": "MyNewSound", "path": "..."}
```

Create with properties:
```python
result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
        "type": "Sound",
        "name": "Footstep_Concrete",
        "@Volume": -6.0,
        "@Pitch": 0,
        "@Lowpass": 0,
        "@IsLoopingEnabled": False
    }
)
```

Create nested hierarchy:
```python
result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
        "type": "RandomSequenceContainer",
        "name": "Footsteps",
        "children": [
            {"type": "Sound", "name": "Footstep_01"},
            {"type": "Sound", "name": "Footstep_02"},
            {"type": "Sound", "name": "Footstep_03"}
        ]
    }
)
```

## Querying Objects with ak.wwise.core.object.get

Retrieve objects from the Wwise project using WAQL or path.

Get by path:
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit"]}
    },
    options={"return": ["id", "name", "type", "path"]}
)
```

Get all children:
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit"]},
        "transform": [{"select": ["children"]}]
    },
    options={"return": ["id", "name", "type", "path"]}
)
```

Get by type (using WAQL):
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"search": [{"type": ["Event"]}]}
    },
    options={"return": ["id", "name", "path"]}
)
```

Get with property values:
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit\\MySound"]}
    },
    options={"return": ["id", "name", "@Volume", "@Pitch", "@Lowpass", "@IsLoopingEnabled"]}
)
```

WAQL query examples:
```python
# Find all sounds with volume below -6dB
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": ["$ from type Sound where Volume < -6"]}
    },
    options={"return": ["id", "name", "@Volume"]}
)

# Find objects by name pattern
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": ['$ from type Sound where name = "Footstep*"']}
    },
    options={"return": ["id", "name", "path"]}
)
```

## Setting Object Properties with ak.wwise.core.object.set

Set single property:

```python

```

## Creating Events

Events link game actions to Wwise playback.

Create a Play event:
```python
# Step 1: Create the Event
event_result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Events\\Default Work Unit",
        "type": "Event",
        "name": "Play_Footstep"
    }
)

# Step 2: Add a Play action targeting a sound
client.call(
    "ak.wwise.core.object.create",
    {
        "parent": event_result["path"],
        "type": "Action",
        "name": "",
        "@ActionType": 1,  # 1 = Play
        "@Target": "\\Actor-Mixer Hierarchy\\Default Work Unit\\Footsteps"
    }
)
```

## Transport Control with ak.wwise.core.transport

Control audio playback in the Wwise Authoring tool.

Create transport and play:
```python
# Create a transport for an object
transport = client.call(
    "ak.wwise.core.transport.create",
    {
        "object": "\\Events\\Default Work Unit\\Play_Footstep"
    }
)

transport_id = transport["transport"]

# Play
client.call(
    "ak.wwise.core.transport.executeAction",
    {
        "transport": transport_id,
        "action": "play"
    }
)

# Stop
client.call(
    "ak.wwise.core.transport.executeAction",
    {
        "transport": transport_id,
        "action": "stop"
    }
)

# Destroy transport when done
client.call(
    "ak.wwise.core.transport.destroy",
    {
        "transport": transport_id
    }
)
```

## Undo/Redo with ak.wwise.core.undo

Manage undo operations:
```python

client.call("ak.wwise.core.undo.beginGroup")

# End undo group with a label
client.call(
    "ak.wwise.core.undo.endGroup",
    {

    }
```

## Subscribing to Topics

WAAPI supports pub/sub for real-time notifications:
```python
# Subscribe to object name changes
def on_name_changed(event):
    print(f"Object renamed: {event}")

handler = client.subscribe(
    "ak.wwise.core.object.nameChanged",
    on_name_changed,
    options={"return": ["id", "name", "path"]}
)

# Subscribe to selection changes
def on_selection_changed(event):
    print(f"Selection: {event}")

handler2 = client.subscribe(
    "ak.wwise.ui.selectionChanged",
    on_selection_changed,
    options={"return": ["id", "name", "type"]}
)

# Unsubscribe
handler.unsubscribe()
handler2.unsubscribe()
```

## Error Handling Best Practices

Robust WAAPI error handling:
```python
from waapi import WaapiClient, CannotConnectToWaapiException

try:
    client = WaapiClient()
except CannotConnectToWaapiException:
    print("Error: Cannot connect to Wwise. Is Wwise running with WAAPI enabled?")
    exit(1)

try:
    result = client.call(
        "ak.wwise.core.object.create",
        {
            "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
            "type": "Sound",
            "name": "TestSound"
        }
    )
    if result:
        print(f"Created: {result['path']}")
    else:
        print("Operation returned no result")
except Exception as e:
    print(f"WAAPI error: {e}")
finally:
    client.disconnect()
```

## Batch Operations Pattern

Efficient batch operations using undo groups:
```python
client = WaapiClient()

# Wrap batch operations in undo group
client.call("ak.wwise.core.undo.beginGroup")

try:
    # Create 50 sound objects
    for i in range(50):
        client.call(
            "ak.wwise.core.object.create",
            {
                "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit\\BatchSounds",
                "type": "Sound",
                "name": f"Sound_{i:03d}",
                "@Volume": -3.0,
                "onNameConflict": "rename"
            }
        )

    client.call(
        "ak.wwise.core.undo.endGroup",
        {"displayName": "Batch Create 50 Sounds"}
    )
    print("Batch creation complete. Can be undone as single action.")

except Exception as e:
    client.call("ak.wwise.core.undo.cancelGroup")
    print(f"Batch failed, all changes rolled back: {e}")

client.disconnect()
```

## Common WAAPI Return Options

Specify what data to return from queries:
```python
# Common return options
options = {
    "return": [
        "id",           # GUID
        "name",         # Object name
        "type",         # Object type (Sound, Event, etc.)
        "path",         # Full path in hierarchy
        "shortId",      # Short ID for runtime
        "classId",      # Class identifier
        "category",     # Object category
        "filePath",     # Source file path
        "workunit",     # Parent Work Unit
        "owner",        # Owner object
        "parent",       # Parent object
        "isPlayable",   # Can be played
        "childrenCount", # Number of children
        "notes",        # Designer notes
        "@Volume",      # Volume property
        "@Pitch",       # Pitch property
        "@Lowpass",     # Low-pass filter
        "@Highpass",    # High-pass filter
        "@IsLoopingEnabled",  # Loop enabled
    ]
}

result = client.call(
    "ak.wwise.core.object.get",
    {"from": {"search": [{"type": ["Sound"]}]}},
    options=options
)
```

## WAQL Advanced Queries

Advanced WAQL query patterns for complex searches:

```python
# Find all events that reference sounds with volume below -10dB
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": [
            '$ from type Event'
        ]}
    },
    options={"return": ["id", "name", "path"]}
)

# Find all Switch Containers
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": [
            '$ from type SwitchContainer'
        ]}
    },
    options={"return": ["id", "name", "path"]}
)

# Find objects modified recently (by name pattern)
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": [
            '$ from type Sound where name : "Weapon"'
        ]}
    },
    options={"return": ["id", "name", "path", "@Volume"]}
)

# Get descendants of a specific path
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit"]},
        "transform": [
            {"select": ["descendants"]},
            {"where": ["type:isIn", ["Sound", "RandomSequenceContainer"]]}
        ]
    },
    options={"return": ["id", "name", "type", "path"]}
)
```
