## WAAPI 连接设置

WAAPI 使用 WAMP（Web Application Messaging Protocol）通过 WebSocket 通信。默认端点：ws://127.0.0.1:8080/waapi

前提条件：
- Wwise 创作应用程序必须正在运行
- WAAPI 必须已启用：Project > Project Settings > WAMP > Enable WAAPI
- 默认端口：8080（可在 Project Settings 中配置）

Python 连接示例（使用 waapi-client）：
```python
from waapi import WaapiClient

# 连接到 Wwise
client = WaapiClient()

# 或指定自定义 URL
client = WaapiClient(url="ws://127.0.0.1:8080/waapi")

# 完成后务必断开连接
client.disconnect()
```

## 使用 ak.wwise.core.object.create 创建对象

在 Wwise 项目层级中创建新对象。

基本创建声音对象：
```python
result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
        "type": "Sound",
        "name": "MyNewSound",
        "onNameConflict": "rename"  # 或 "replace"、"fail"
    }
)
# 返回: {"id": "{GUID}", "name": "MyNewSound", "path": "..."}
```

带属性创建：
```python
result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
        "type": "Sound",
        "name": "Footstep_Concrete",
        "@Volume": -6.0,
        "@Pitch": 0,
        "@Lowpass": 0,
        "@IsLoopingEnabled": False
    }
)
```

创建嵌套层级结构：
```python
result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
        "type": "RandomSequenceContainer",
        "name": "Footsteps",
        "children": [
            {"type": "Sound", "name": "Footstep_01"},
            {"type": "Sound", "name": "Footstep_02"},
            {"type": "Sound", "name": "Footstep_03"}
        ]
    }
)
```

## 使用 ak.wwise.core.object.get 查询对象

从 Wwise 项目中检索对象，支持 WAQL 和路径查询。

通过路径查询：
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit"]}
    },
    options={"return": ["id", "name", "type", "path"]}
)
```

获取所有子对象：
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit"]},
        "transform": [{"select": ["children"]}]
    },
    options={"return": ["id", "name", "type", "path"]}
)
```

按类型查询（使用 WAQL）：
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"search": [{"type": ["Event"]}]}
    },
    options={"return": ["id", "name", "path"]}
)
```

查询并返回属性值：
```python
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"path": ["\\Actor-Mixer Hierarchy\\Default Work Unit\\MySound"]}
    },
    options={"return": ["id", "name", "@Volume", "@Pitch", "@Lowpass", "@IsLoopingEnabled"]}
)
```

WAQL 查询示例：
```python
# 查找所有音量低于 -6dB 的 Sound
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": ["$ from type Sound where Volume < -6"]}
    },
    options={"return": ["id", "name", "@Volume"]}
)

# 按名称模式查找对象
result = client.call(
    "ak.wwise.core.object.get",
    {
        "from": {"query": ['$ from type Sound where name = "Footstep*"']}
    },
    options={"return": ["id", "name", "path"]}
)
```

## 创建 Event

Event 将游戏动作与 Wwise 播放关联起来。

创建 Play 事件：
```python
# 第1步：创建 Event
event_result = client.call(
    "ak.wwise.core.object.create",
    {
        "parent": "\\Events\\Default Work Unit",
        "type": "Event",
        "name": "Play_Footstep"
    }
)

# 第2步：添加一个 Play Action，目标指向一个声音对象
client.call(
    "ak.wwise.core.object.create",
    {
        "parent": event_result["path"],
        "type": "Action",
        "name": "",
        "@ActionType": 1,  # 1 = Play
        "@Target": "\\Actor-Mixer Hierarchy\\Default Work Unit\\Footsteps"
    }
)
```

## 使用 ak.wwise.core.transport 控制传输

在 Wwise 创作工具中控制音频播放。

创建传输并播放：
```python
# 为对象创建传输控制实例
transport = client.call(
    "ak.wwise.core.transport.create",
    {
        "object": "\\Events\\Default Work Unit\\Play_Footstep"
    }
)

transport_id = transport["transport"]

# 播放
client.call(
    "ak.wwise.core.transport.executeAction",
    {
        "transport": transport_id,
        "action": "play"
    }
)

# 停止
client.call(
    "ak.wwise.core.transport.executeAction",
    {
        "transport": transport_id,
        "action": "stop"
    }
)

# 完成后销毁传输
client.call(
    "ak.wwise.core.transport.destroy",
    {
        "transport": transport_id
    }
)
```

## 订阅主题事件

WAAPI 支持发布/订阅模式实现实时通知：
```python
# 订阅对象名称变更
def on_name_changed(event):
    print(f"对象已重命名: {event}")

handler = client.subscribe(
    "ak.wwise.core.object.nameChanged",
    on_name_changed,
    options={"return": ["id", "name", "path"]}
)

# 订阅选择变更
def on_selection_changed(event):
    print(f"当前选择: {event}")

handler2 = client.subscribe(
    "ak.wwise.ui.selectionChanged",
    on_selection_changed,
    options={"return": ["id", "name", "type"]}
)

# 取消订阅
handler.unsubscribe()
handler2.unsubscribe()
```

## 错误处理最佳实践

稳健的 WAAPI 错误处理：
```python
from waapi import WaapiClient, CannotConnectToWaapiException

try:
    client = WaapiClient()
except CannotConnectToWaapiException:
    print("错误：无法连接到 Wwise。请确认 Wwise 正在运行且已启用 WAAPI。")
    exit(1)

try:
    result = client.call(
        "ak.wwise.core.object.create",
        {
            "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit",
            "type": "Sound",
            "name": "TestSound"
        }
    )
    if result:
        print(f"已创建: {result['path']}")
    else:
        print("操作未返回结果")
except Exception as e:
    print(f"WAAPI 错误: {e}")
finally:
    client.disconnect()
```

## 批量操作模式

使用撤销组进行高效批量操作：
```python
client = WaapiClient()

# 将批量操作包装在撤销组中
client.call("ak.wwise.core.undo.beginGroup")

try:
    # 创建 50 个声音对象
    for i in range(50):
        client.call(
            "ak.wwise.core.object.create",
            {
                "parent": "\\Actor-Mixer Hierarchy\\Default Work Unit\\BatchSounds",
                "type": "Sound",
                "name": f"Sound_{i:03d}",
                "@Volume": -3.0,
                "onNameConflict": "rename"
            }
        )

    client.call(
        "ak.wwise.core.undo.endGroup",
        {"displayName": "批量创建 50 个声音对象"}
    )
    print("批量创建完成。可作为单次操作撤销。")

except Exception as e:
    client.call("ak.wwise.core.undo.cancelGroup")
    print(f"批量操作失败，所有更改已回滚: {e}")

client.disconnect()
```

## 常用 WAAPI 返回选项

指定查询返回的数据字段：
```python
# 常用返回选项
options = {
    "return": [
        "id",            # GUID 唯一标识符
        "name",          # 对象名称
        "type",          # 对象类型（Sound、Event 等）
        "path",          # 在层级中的完整路径
        "shortId",       # 运行时短 ID
        "classId",       # 类标识符
        "category",      # 对象分类
        "filePath",      # 源文件路径
        "workunit",      # 所属 Work Unit
        "owner",         # 所有者对象
        "parent",        # 父对象
        "isPlayable",    # 是否可播放
        "childrenCount", # 子对象数量
        "notes",         # 设计师备注
        "@Volume",       # 音量属性
        "@Pitch",        # 音高属性
        "@Lowpass",      # 低通滤波器
        "@Highpass",     # 高通滤波器
        "@IsLoopingEnabled",  # 是否启用循环
    ]
}

result = client.call(
    "ak.wwise.core.object.get",
    {"from": {"search": [{"type": ["Sound"]}]}},
    options=options
)
```
